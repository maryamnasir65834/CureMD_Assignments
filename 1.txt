class MainClass
{
    public static int BracketMatcher(string str)
    {
       
        int CountForOpenBrackets = 0;
        int CountForCloseBrackets = 0;

        foreach (char c in str)
        {
            if (c == '(')
            {
                CountForOpenBrackets++;
            }
            else if (c == ')')
            {
                CountForCloseBrackets++;

        
                if (CountForCloseBrackets > CountForOpenBrackets)
                {
                    return 0;
                }
            }
        }

        if (CountForOpenBrackets != CountForCloseBrackets)
        {
            return 0;
        }


        if (CountForOpenBrackets == 0 && CountForCloseBrackets == 0)
        {
            return 1;
        }

      
        return 1;
    }

    public static void Main(string[] args)
    {
        // Example usage:
        Console.WriteLine(BracketMatcher("(hello(world))")); // Output: 0
        Console.WriteLine(BracketMatcher("(hello(world)"));  // Output: 0
        Console.WriteLine(BracketMatcher("hello(world)"));   // Output: 1
        Console.WriteLine(BracketMatcher(""));               // Output: 1
        Console.WriteLine(BracketMatcher("()"));             // Output: 1
        Console.WriteLine(BracketMatcher(")("));
        Console.WriteLine(BracketMatcher(""));
        // Output: 0
    }
}


using System;
using System.Diagnostics.CodeAnalysis;

class MainClass
{
    public static string Superincreasing(int[] arr)
    {
        
        int sum = 0;


        foreach (int num in arr)
        {
         
            if (num <= sum)
            {
                return "false";
            }
        
            sum += num;
        }

       
        return "true";
    }

    public static void Main(string[] args)
    {
        int[] arr1 = { 1, 3, 6, 13, 55 };
        Console.WriteLine(Superincreasing(arr1)); 

        int[] arr2 = { 1, 2, 4, 8, 6 };
        Console.WriteLine(Superincreasing(arr2)); 
    }
}
