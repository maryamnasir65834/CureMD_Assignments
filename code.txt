using System;

class Program
{
    // Question no3(Checking LEAP YEAR)
    static void LEAP_YEAR(int year, int no_of_days)
    {
      

        if (no_of_days == 366)
            {
            Console.WriteLine($"This is a Leap Year: {year}");
            }
            else
            {
                Console.WriteLine($"This year {year} is not a Leap Year");
            }
       
    }
    // Question no4 (Distance conversion
    static double DistanceConversion(double Km_per_hour)
    {
        double Miles_per_hour = Km_per_hour * 0.621371;
        Console.WriteLine($"Converted speed is: {Miles_per_hour} miles per hour");
        return Miles_per_hour;
    }
    static void Main()
    {
        // Question no 1(Summing of Two Numbers)
        int A, B, C;

        Console.Write("Enter First Number:");
        A = int.Parse(Console.ReadLine());

        Console.Write("Enter Second Number:");
        B = int.Parse(Console.ReadLine());

        C = A + B;

        Console.WriteLine("Sum of two Numbers = " + C);

        //Question no 2(Even Numbers)

        int Number=1;
        while (Number <= 100)
        {
            if (Number % 2 == 0)
            {
                Console.WriteLine("Even Numbers are :" + Number);
            }
            else
            {
                Number = Number + 1;
                continue;
            }
            Number = Number + 1;

        }
        //Question no 3(Checking for Leap Year)
        Console.WriteLine("Enter the year:");
        int year = int.Parse(Console.ReadLine()); // Input year from the user

        Console.WriteLine("Enter the number of days (365 or 366):");
        int no_of_days = int.Parse(Console.ReadLine()); // Input number of days from the user

        LEAP_YEAR(year, no_of_days);

        // Question no4 (Distance conversion

        Console.WriteLine("Enter speed in kilometers per hour:");
        double Km_per_hour = double.Parse(Console.ReadLine()); // Input speed from the user

        double result = DistanceConversion(Km_per_hour);

    }
}
